633  int inc_forkcount(void) {
634  if (0 == forkmax || 0 == shmid)
635  return 0;
636  int original;
637  int new_value;
638 
639  do {
640  original = *forkcurr;
641  new_value = original + 1;
642  __asm(" cs %0,%2,%1 \n "
643  : "+r"(original), "+m"(*forkcurr)
644  : "r"(new_value)
645  :);
646  } while (original != (new_value - 1));
647  return new_value;
648  }
649  int dec_forkcount(void) {
650  if (0 == forkmax || 0 == shmid)
651  return 0;
652  int original;
653  int new_value;
654 
655  do {
656  original = *forkcurr;
657  if (original == 0)
658  return 0;
659  new_value = original - 1;
660  __asm(" cs %0,%2,%1 \n "
661  : "+r"(original), "+m"(*forkcurr)
662  : "r"(new_value)
663  :);
664  } while (original != (new_value - 1));
665  return new_value;
666  }
667  int shmid_value(void) { return shmid; }

669  ~__zinit() {
670  if (_CVTSTATE_OFF == cvstate) {
671  __ae_autoconvert_state(cvstate);
672  }
673  __ae_thread_swapmode(mode);
674  if (shmid != 0) {
675  if (__forked)
676  dec_forkcount();
677  shmdt(forkcurr);
678  shmctl(shmid, IPC_RMID, 0);
679  }
680  __cleanupipc(0);
681  }
