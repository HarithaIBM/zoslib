650  int inc_forkcount(void) {
651  if (0 == forkmax || 0 == shmid)
652  return 0;
653  int original;
654  int new_value;
655 
656  do {
657  original = *forkcurr;
658  new_value = original + 1;
659  __asm(" cs %0,%2,%1 \n "
660  : "+r"(original), "+m"(*forkcurr)
661  : "r"(new_value)
662  :);
663  } while (original != (new_value - 1));
664  return new_value;
665  }
666  int dec_forkcount(void) {
667  if (0 == forkmax || 0 == shmid)
668  return 0;
669  int original;
670  int new_value;
671 
672  do {
673  original = *forkcurr;
674  if (original == 0)
675  return 0;
676  new_value = original - 1;
677  __asm(" cs %0,%2,%1 \n "
678  : "+r"(original), "+m"(*forkcurr)
679  : "r"(new_value)
680  :);
681  } while (original != (new_value - 1));
682  return new_value;
683  }
684  int shmid_value(void) { return shmid; }
685 
686  ~__zinit() {
687  if (_CVTSTATE_OFF == cvstate) {
688  __ae_autoconvert_state(cvstate);
689  }
690  __ae_thread_swapmode(mode);
691  if (shmid != 0) {
692  if (__forked)
693  dec_forkcount();
694  shmdt(forkcurr);
695  shmctl(shmid, IPC_RMID, 0);
696  }
697  __cleanupipc(0);
698  }
699  void __abort() { _th(); }
