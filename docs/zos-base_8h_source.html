803  int inc_forkcount(void) {
804  if (0 == forkmax || 0 == shmid)
805  return 0;
806  int original;
807  int new_value;
808 
809  do {
810  original = *forkcurr;
811  new_value = original + 1;
812  __asm(" cs %0,%2,%1 \n "
813  : "+r"(original), "+m"(*forkcurr)
814  : "r"(new_value)
815  :);
816  } while (original != (new_value - 1));
817  return new_value;
818  }
819  int dec_forkcount(void) {
820  if (0 == forkmax || 0 == shmid)
821  return 0;
822  int original;
823  int new_value;
824 
825  do {
826  original = *forkcurr;
827  if (original == 0)
828  return 0;
829  new_value = original - 1;
830  __asm(" cs %0,%2,%1 \n "
831  : "+r"(original), "+m"(*forkcurr)
832  : "r"(new_value)
833  :);
834  } while (original != (new_value - 1));
835  return new_value;
836  }
837  int shmid_value(void) { return shmid; }
838 
839  ~__zinit() {
840  if (_CVTSTATE_OFF == cvstate) {
841  __ae_autoconvert_state(cvstate);
842  }
843  __ae_thread_swapmode(mode);
844  if (shmid != 0) {
845  if (__forked)
846  dec_forkcount();
847  shmdt(forkcurr);
848  shmctl(shmid, IPC_RMID, 0);
849  }
850  __cleanupipc(0);
851  }
